# Data Visualization with visx

**Low-level visualization primitives for React**

---

## Why visx?

- **Composable** — Build exactly what you need
- **Performant** — Renders to SVG, no canvas overhead
- **React-native** — First-class React components
- **Type-safe** — Full TypeScript support
- **Unopinionated** — No predefined chart types

## Bar Chart

export const BarChart = () => {
  const data = [
    { label: 'Mon', value: 42 },
    { label: 'Tue', value: 65 },
    { label: 'Wed', value: 38 },
    { label: 'Thu', value: 78 },
    { label: 'Fri', value: 56 },
    { label: 'Sat', value: 89 },
    { label: 'Sun', value: 71 },
  ]
  const width = 500, height = 300, margin = { top: 20, right: 20, bottom: 40, left: 40 }
  const xMax = width - margin.left - margin.right
  const yMax = height - margin.top - margin.bottom
  const xScale = scaleBand({ domain: data.map(d => d.label), range: [0, xMax], padding: 0.3 })
  const yScale = scaleLinear({ domain: [0, Math.max(...data.map(d => d.value))], range: [yMax, 0], nice: true })
  return (
    <svg width={width} height={height}>
      <Group left={margin.left} top={margin.top}>
        <GridRows scale={yScale} width={xMax} stroke="currentColor" strokeOpacity={0.1} />
        {data.map(d => (
          <Bar key={d.label} x={xScale(d.label)} y={yScale(d.value)} width={xScale.bandwidth()} height={yMax - yScale(d.value)} fill="currentColor" fillOpacity={0.8} />
        ))}
        <AxisBottom top={yMax} scale={xScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'middle' }} />
        <AxisLeft scale={yScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'end', dx: -4 }} />
      </Group>
    </svg>
  )
}

<div className="flex justify-center">
  <BarChart />
</div>

## Line Chart

export const LineChart = () => {
  const data = Array.from({ length: 20 }, (_, i) => ({
    x: i,
    y: Math.sin(i * 0.5) * 30 + 50 + Math.random() * 10
  }))
  const width = 500, height = 300, margin = { top: 20, right: 20, bottom: 40, left: 40 }
  const xMax = width - margin.left - margin.right
  const yMax = height - margin.top - margin.bottom
  const xScale = scaleLinear({ domain: [0, 19], range: [0, xMax] })
  const yScale = scaleLinear({ domain: [0, 100], range: [yMax, 0] })
  return (
    <svg width={width} height={height}>
      <Group left={margin.left} top={margin.top}>
        <GridRows scale={yScale} width={xMax} stroke="currentColor" strokeOpacity={0.1} />
        <GridColumns scale={xScale} height={yMax} stroke="currentColor" strokeOpacity={0.1} />
        <LinePath data={data} x={d => xScale(d.x)} y={d => yScale(d.y)} stroke="currentColor" strokeWidth={1.5} curve={curveMonotoneX} />
        {data.map((d, i) => (
          <Circle key={i} cx={xScale(d.x)} cy={yScale(d.y)} r={3} fill="currentColor" />
        ))}
        <AxisBottom top={yMax} scale={xScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'middle' }} />
        <AxisLeft scale={yScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'end', dx: -4 }} />
      </Group>
    </svg>
  )
}

<div className="flex justify-center">
  <LineChart />
</div>

export const AreaChart = () => {

## Interactive Chart

export const InteractiveChart = () => {
  const [hoveredBar, setHoveredBar] = useState(null)
  const data = [
    { label: 'React', value: 85 },
    { label: 'Vue', value: 62 },
    { label: 'Angular', value: 45 },
    { label: 'Svelte', value: 38 },
    { label: 'Solid', value: 28 },
  ]
  const width = 500, height = 300, margin = { top: 20, right: 20, bottom: 50, left: 50 }
  const xMax = width - margin.left - margin.right
  const yMax = height - margin.top - margin.bottom
  const xScale = scaleBand({ domain: data.map(d => d.label), range: [0, xMax], padding: 0.3 })
  const yScale = scaleLinear({ domain: [0, 100], range: [yMax, 0] })
  return (
    <svg width={width} height={height}>
      <Group left={margin.left} top={margin.top}>
        <GridRows scale={yScale} width={xMax} stroke="currentColor" strokeOpacity={0.1} />
        {data.map(d => (
          <g key={d.label}>
            <Bar
              x={xScale(d.label)}
              y={yScale(d.value)}
              width={xScale.bandwidth()}
              height={yMax - yScale(d.value)}
              fill="currentColor"
              opacity={hoveredBar === null || hoveredBar === d.label ? 0.8 : 0.2}
              onMouseEnter={() => setHoveredBar(d.label)}
              onMouseLeave={() => setHoveredBar(null)}
              style={{ cursor: 'pointer', transition: 'opacity 0.2s' }}
            />
            {hoveredBar === d.label && (
              <Text x={xScale(d.label) + xScale.bandwidth() / 2} y={yScale(d.value) - 8} textAnchor="middle" fill="currentColor" fontSize={14} fontWeight="bold">
                {d.value}%
              </Text>
            )}
          </g>
        ))}
        <AxisBottom top={yMax} scale={xScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'middle' }} />
        <AxisLeft scale={yScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickFormat={v => `${v}%`} tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'end', dx: -4 }} />
      </Group>
    </svg>
  )
}

<div className="flex justify-center">
  <InteractiveChart />
</div>

## Multi-Series

export const MultiSeriesChart = () => {
  const series = [
    { name: 'Revenue', opacity: 1, dash: '', data: [30, 45, 55, 70, 65, 80, 95] },
    { name: 'Users', opacity: 0.6, dash: '4,4', data: [20, 35, 40, 50, 60, 75, 85] },
    { name: 'Sessions', opacity: 0.35, dash: '2,2', data: [10, 25, 35, 40, 45, 55, 60] },
  ]
  const width = 500, height = 300, margin = { top: 20, right: 100, bottom: 40, left: 50 }
  const xMax = width - margin.left - margin.right
  const yMax = height - margin.top - margin.bottom
  const xScale = scaleLinear({ domain: [0, 6], range: [0, xMax] })
  const yScale = scaleLinear({ domain: [0, 100], range: [yMax, 0] })
  return (
    <svg width={width} height={height}>
      <Group left={margin.left} top={margin.top}>
        <GridRows scale={yScale} width={xMax} stroke="currentColor" strokeOpacity={0.1} />
        {series.map(s => (
          <g key={s.name}>
            <LinePath data={s.data.map((y, x) => ({ x, y }))} x={d => xScale(d.x)} y={d => yScale(d.y)} stroke="currentColor" strokeOpacity={s.opacity} strokeWidth={1.5} strokeDasharray={s.dash} curve={curveMonotoneX} />
            {s.data.map((y, x) => (
              <Circle key={x} cx={xScale(x)} cy={yScale(y)} r={3} fill="currentColor" fillOpacity={s.opacity} />
            ))}
          </g>
        ))}
        <AxisBottom top={yMax} scale={xScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickFormat={d => ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'][d]} tickLabelProps={{ fill: 'currentColor', fontSize: 11, textAnchor: 'middle' }} />
        <AxisLeft scale={yScale} stroke="currentColor" strokeOpacity={0.3} tickStroke="currentColor" tickLabelProps={{ fill: 'currentColor', fontSize: 12, textAnchor: 'end', dx: -4 }} />
        {series.map((s, i) => (
          <g key={s.name} transform={`translate(${xMax + 15}, ${i * 22 + 10})`}>
            <line x1={0} y1={6} x2={12} y2={6} stroke="currentColor" strokeOpacity={s.opacity} strokeWidth={2} strokeDasharray={s.dash} />
            <Text x={18} y={10} fill="currentColor" fontSize={12}>{s.name}</Text>
          </g>
        ))}
      </Group>
    </svg>
  )
}

<div className="flex justify-center">
  <MultiSeriesChart />
</div>
